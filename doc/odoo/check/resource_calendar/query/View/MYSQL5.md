```sql
SELECT
t1.`ACTIVE`,
t1.`COMPANY_ID`,
t11.`NAME` AS `COMPANY_NAME`,
t1.`CREATE_DATE`,
t1.`CREATE_UID`,
t1.`FLEXIBLE_HOURS`,
t1.`FULL_TIME_REQUIRED_HOURS`,
```
